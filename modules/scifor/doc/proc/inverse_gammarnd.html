function inverse_gammarnd(shape,scale) result(ans)
  real(8) :: shape,scale,ans
  ! If X is gamma(shape, scale) then
  ! 1/Y is inverse gamma(shape, 1/scale)
  ans= 1d0/gammarnd(shape,1d0/scale)
end function inverse_gammarnd
